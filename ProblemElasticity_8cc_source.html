92 Real ProblemElasticity::computeGradient(EquationSystems & stateAdj, const Point & p) const
93 {
94  Gradient du = stateAdj.get_system(name_).point_gradient(0, p);
95  Gradient dv = stateAdj.get_system(name_).point_gradient(1, p);
96 
97  Real sum;
98  sum = -2. * coeff_mu_ * (
99  du(0) * du(0) + dv(1) * dv(1) +
100  (
101  (dv(1) + du(0)) *
102  (dv(1) + du(0))
103  ) / 4 +
104  (
105  (du(1) + dv(0)) *

107  ) / 4.) - coeff_lambda_ * (du(0) + dv(1)) * (du(0) + dv(1));
